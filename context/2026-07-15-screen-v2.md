# 2026-07-15 screen.py 新增 heat 资金热度入口

## 原因

需要在 `screen.py` 中新增一个资金热度选股入口，用来从全部 A 股中筛出当日资金活跃、量价同步、但不做复杂排序的候选股票。

第一版 `heat` 只做硬过滤，不做排序权重、`screen_score`、风险扣分、主题热度和 alphasift 的复杂评分逻辑。

入口名称确认使用：

```powershell
python screen.py heat
```

## 技术方案

第一版 `heat` 复用现有 `screen.py` 和 `data.py` 能复用的能力：

1. 复用 `get_all_a_share_codes()` 获取全部 A 股股票池。
2. 复用 `data.load_breakout_kline()` 获取 K 线数据。
3. 复用 `run_layer()` 输出每一层的输入数量、通过数量和淘汰数量。
4. 默认只输出结果摘要，传入 `--debug` 时输出完整 JSON。

`data.load_breakout_kline()` 当前已经包含盘中/盘外逻辑：

```text
screen.py
  -> data.load_breakout_kline()
    -> data.load_daily_kline()
      -> load_daily_kline_rows_from_db()
      -> 如果盘中，再 load_realtime_daily_kline()
      -> merge_realtime_kline_rows()
```

当前实时 K 线来源是：

```python
get_market_data(period="1d", count=1)
```

第一版 `heat` 不改数据层，不把实时 K 线来源改成 snapshot。

硬过滤条件：

```yaml
exclude_st: true
amount_min: 30000
turnover_rate_min: 2.0
volume_ratio_min: 1.5
change_pct_min: 1.0
price_min: 3
price_max: 220
```

字段口径：

1. `amount_min` 使用万元口径，原始参考条件 `300000000` 元在 Cassa 中改为 `30000` 万元，直接使用 K 线 `amount` 字段判断，不改数据。
2. `price` 使用最新 K 线 `close_price`。
3. `change_pct` 使用最新 K 线 `close_price / 前一根 close_price - 1` 后换算百分比。
4. `volume_ratio` 使用最新 K 线 `volume / 前 5 根 K 线平均 volume`。
5. `turnover_rate` 使用 `data.get_more_info(code)["fHSL"]`。
6. `exclude_st` 优先使用 `data.get_stock_info(code)["Name"]` 判断，名称中包含 `ST` 或 `退` 的股票剔除。

分层流程：

```text
初始股票池
ST/退市过滤
价格区间过滤
成交额过滤
换手率过滤
量比过滤
涨幅过滤
最终入选
```

`selected_items` 建议保留字段：

```text
code
name
trade_date
price
amount
turnover_rate
volume_ratio
change_pct
```

```python
def safe_float(value, default=0.0):
    """安全转换 float。"""
    try:
        if value is None:
            return default
        return float(value)
    except (TypeError, ValueError):
        return default


def build_heat_metric_map(stock_codes, kline_map):
    """根据 K 线构造 heat 硬过滤所需指标。"""
    metric_map = {}

    for stock_code in stock_codes:
        kline_bars = kline_map.get(stock_code, [])
        if len(kline_bars) < 2:
            continue

        latest_kline = kline_bars[-1]
        previous_kline = kline_bars[-2]
        price = safe_float(latest_kline.get("close_price"))
        previous_close = safe_float(previous_kline.get("close_price"))
        change_pct = (
            (price / previous_close - 1) * 100
            if price > 0 and previous_close > 0
            else 0.0
        )

        previous_rows = kline_bars[-6:-1]
        previous_volumes = [
            safe_float(row.get("volume"))
            for row in previous_rows
            if safe_float(row.get("volume")) > 0
        ]
        average_volume = (
            sum(previous_volumes) / len(previous_volumes)
            if len(previous_volumes) >= 5
            else 0.0
        )
        latest_volume = safe_float(latest_kline.get("volume"))
        volume_ratio = (
            latest_volume / average_volume
            if latest_volume > 0 and average_volume > 0
            else 0.0
        )

        metric_map[stock_code] = {
            "code": stock_code,
            "trade_date": latest_kline.get("trade_date", ""),
            "price": round(price, 4),
            "amount": round(safe_float(latest_kline.get("amount")), 4),
            "change_pct": round(change_pct, 4),
            "volume_ratio": round(volume_ratio, 6),
        }

    return metric_map
```

```python
def enrich_heat_realtime_info(metric_map):
    """补充股票名称和换手率。"""
    for stock_code, metrics in metric_map.items():
        stock_info = data.get_stock_info(stock_code, field_list=[])
        more_info = data.get_more_info(stock_code, field_list=[])

        name = ""
        if isinstance(stock_info, dict):
            name = str(stock_info.get("Name", "") or "").strip()
        if not name and isinstance(more_info, dict):
            name = str(more_info.get("Name", "") or "").strip()

        metrics["name"] = name
        metrics["turnover_rate"] = round(
            safe_float((more_info or {}).get("fHSL")),
            4,
        )

    return metric_map
```

```python
def filter_heat_exclude_st(stock_codes, metric_map):
    """剔除 ST / 退市相关股票。"""
    passed_codes = []
    for stock_code in stock_codes:
        name = str(metric_map.get(stock_code, {}).get("name", ""))
        if "ST" in name.upper() or "退" in name:
            continue
        passed_codes.append(stock_code)
    return passed_codes


def filter_heat_price_range(
    stock_codes,
    metric_map,
    price_min=DEFAULT_HEAT_PRICE_MIN,
    price_max=DEFAULT_HEAT_PRICE_MAX,
):
    """筛选价格区间。"""
    return [
        stock_code
        for stock_code in stock_codes
        if float(price_min) <= safe_float(metric_map.get(stock_code, {}).get("price")) <= float(price_max)
    ]


def filter_heat_amount(
    stock_codes,
    metric_map,
    amount_min=DEFAULT_HEAT_AMOUNT_MIN,
):
    """筛选成交额，amount_min 使用万元口径。"""
    return [
        stock_code
        for stock_code in stock_codes
        if safe_float(metric_map.get(stock_code, {}).get("amount")) >= float(amount_min)
    ]


def filter_heat_turnover_rate(
    stock_codes,
    metric_map,
    turnover_rate_min=DEFAULT_HEAT_TURNOVER_RATE_MIN,
):
    """筛选换手率。"""
    return [
        stock_code
        for stock_code in stock_codes
        if safe_float(metric_map.get(stock_code, {}).get("turnover_rate")) >= float(turnover_rate_min)
    ]


def filter_heat_volume_ratio(
    stock_codes,
    metric_map,
    volume_ratio_min=DEFAULT_HEAT_VOLUME_RATIO_MIN,
):
    """筛选量比。"""
    return [
        stock_code
        for stock_code in stock_codes
        if safe_float(metric_map.get(stock_code, {}).get("volume_ratio")) >= float(volume_ratio_min)
    ]


def filter_heat_change_pct(
    stock_codes,
    metric_map,
    change_pct_min=DEFAULT_HEAT_CHANGE_PCT_MIN,
):
    """筛选涨幅。"""
    return [
        stock_code
        for stock_code in stock_codes
        if safe_float(metric_map.get(stock_code, {}).get("change_pct")) >= float(change_pct_min)
    ]
```

```python
def screen_heat(
    heat_date="",
    amount_min=DEFAULT_HEAT_AMOUNT_MIN,
    turnover_rate_min=DEFAULT_HEAT_TURNOVER_RATE_MIN,
    volume_ratio_min=DEFAULT_HEAT_VOLUME_RATIO_MIN,
    change_pct_min=DEFAULT_HEAT_CHANGE_PCT_MIN,
    price_min=DEFAULT_HEAT_PRICE_MIN,
    price_max=DEFAULT_HEAT_PRICE_MAX,
    batch_size=DEFAULT_BATCH_SIZE,
):
    """从全部 A 股中筛选资金热度候选。"""
    all_stock_codes = get_all_a_share_codes()
    if not all_stock_codes:
        raise RuntimeError("未获取到全部 A 股股票列表")

    layer_records = []
    print(f"[选股] 初始股票池：{len(all_stock_codes)}")

    kline_map = data.load_breakout_kline(
        stock_list=all_stock_codes,
        box_days=DEFAULT_HEAT_KLINE_DAYS - 1,
        breakout_date=heat_date,
        batch_size=batch_size,
    )
    metric_map = build_heat_metric_map(all_stock_codes, kline_map)
    metric_map = enrich_heat_realtime_info(metric_map)
    available_codes = [
        stock_code
        for stock_code in all_stock_codes
        if stock_code in metric_map
    ]

    current_codes = run_layer(
        layer_name="ST/退市过滤",
        input_codes=available_codes,
        filter_func=lambda codes: filter_heat_exclude_st(codes, metric_map),
        layer_records=layer_records,
    )
    current_codes = run_layer(
        layer_name="价格区间过滤",
        input_codes=current_codes,
        filter_func=lambda codes: filter_heat_price_range(
            codes,
            metric_map,
            price_min=price_min,
            price_max=price_max,
        ),
        layer_records=layer_records,
    )
    current_codes = run_layer(
        layer_name="成交额过滤",
        input_codes=current_codes,
        filter_func=lambda codes: filter_heat_amount(
            codes,
            metric_map,
            amount_min=amount_min,
        ),
        layer_records=layer_records,
    )
    current_codes = run_layer(
        layer_name="换手率过滤",
        input_codes=current_codes,
        filter_func=lambda codes: filter_heat_turnover_rate(
            codes,
            metric_map,
            turnover_rate_min=turnover_rate_min,
        ),
        layer_records=layer_records,
    )
    current_codes = run_layer(
        layer_name="量比过滤",
        input_codes=current_codes,
        filter_func=lambda codes: filter_heat_volume_ratio(
            codes,
            metric_map,
            volume_ratio_min=volume_ratio_min,
        ),
        layer_records=layer_records,
    )
    final_codes = run_layer(
        layer_name="涨幅过滤",
        input_codes=current_codes,
        filter_func=lambda codes: filter_heat_change_pct(
            codes,
            metric_map,
            change_pct_min=change_pct_min,
        ),
        layer_records=layer_records,
    )

    print(f"[选股] 最终入选：{len(final_codes)}")

    selected_items = [
        dict(metric_map.get(stock_code, {}))
        for stock_code in final_codes
    ]

    return {
        "strategy": "heat",
        "heat_date": heat_date or "",
        "amount_min": float(amount_min),
        "turnover_rate_min": float(turnover_rate_min),
        "volume_ratio_min": float(volume_ratio_min),
        "change_pct_min": float(change_pct_min),
        "price_min": float(price_min),
        "price_max": float(price_max),
        "initial_count": len(all_stock_codes),
        "available_count": len(available_codes),
        "selected_count": len(final_codes),
        "selected_codes": final_codes,
        "selected_items": selected_items,
        "layers": layer_records,
    }
```

# 2026-07-15 screen.py 全局选股耗时日志

## 原因

当前 `screen.py` 的选股流程在执行全市场扫描时可能耗时较久，尤其是盘中模式下会触发实时 K 线合并。

现有日志只能看到类似：

```text
[选股] 初始股票池：5535
[数据] 突破日期：2026-07-15
[数据] 当前模式：盘中
```

如果后续长时间没有输出，无法判断程序是在读取本地 DB、拉取通达信实时 K 线、合并 K 线、执行某层过滤，还是组装结果。

因此需要给所有选股入口增加统一耗时日志，而不是只针对 `heat`。

## 技术方案

本次改动只增加可观测性，不改变任何选股逻辑、数据来源、盘中拼接规则和过滤条件。

覆盖范围：

1. `screen_box_consolidation`
2. `screen_volume_breakout`
3. `screen_breakout_pullback_ma`
4. `screen_heat`
5. 后续新增的其他选股入口

核心思路：

1. 在 `screen.py` 中新增通用计时工具。
2. 改造 `run_layer()`，让每一层筛选自动打印耗时，并把耗时写入 `layer_records`。
3. 给各个 `screen_*` 主流程的大阶段增加耗时日志，例如获取股票池、读取 K 线、结果组装、全流程完成。
4. 在 `data.py` 的共用数据读取函数中增加耗时和批次进度日志，尤其是 `load_realtime_daily_kline()`。

耗时日志默认开启。

`--debug` 仍然只控制完整 JSON 输出，不控制耗时日志。

建议日志示例：


### screen.py 新增计时工具

```python
def elapsed_seconds(started_at):
    """返回从 started_at 到当前的秒数。"""
    return round(time.perf_counter() - started_at, 1)


@contextmanager
def timed_step(label):
    """打印一个阶段的开始和结束耗时。"""
    started_at = time.perf_counter()
    print(f"[耗时] {label} 开始")
    try:
        yield
    finally:
        print(f"[耗时] {label} 完成：{elapsed_seconds(started_at)}s")
```

### screen.py 修改 run_layer

替换原函数：

```python
def run_layer(layer_name, input_codes, filter_func, layer_records):
    """执行一层筛选并记录通过数量和耗时。"""
    started_at = time.perf_counter()
    before_count = len(input_codes)
    output_codes = filter_func(input_codes)
    after_count = len(output_codes)
    removed_count = before_count - after_count
    elapsed = elapsed_seconds(started_at)

    print(
        f"[选股] {layer_name}: "
        f"输入 {before_count}，通过 {after_count}，淘汰 {removed_count}，"
        f"耗时 {elapsed}s"
    )

    layer_records.append(
        {
            "layer": layer_name,
            "input_count": before_count,
            "passed_count": after_count,
            "removed_count": removed_count,
            "elapsed_seconds": elapsed,
        }
    )

    return output_codes
```

# 2026-07-15 screen.py 统一选股 JSON 输出

## 原因

当前不同选股策略的 JSON 输出结构不一致：放量突破会把箱体和突破字段直接平铺在 `selected_items` 中，资金热度会把价格、成交额、涨幅、量比、换手率等字段直接平铺在 `selected_items` 中。这样后续如果要统一展示、统一保存、统一给其他模块消费，会出现每个策略都要单独适配的问题。

目标是把所有选股策略统一为同一种 JSON 外壳：

```json
{
  "strategy": "heat",
  "run_date": "2026-07-15",
  "mode": "盘中",
  "conditions": [],
  "selected": [
    {
      "code": "000554.SZ",
      "name": "泰山石油",
      "sectors": [
        {
          "type": "industry",
          "name": "XXX",
          "code": ""
        }
      ],
      "latest_kline": {}
    }
  ],
  "layers": []
}
```

其中：

- `conditions` 表示策略条件定义，需要和 `layers` 的筛选层级对应。
- `layers` 表示每一层筛选的执行结果，包括输入数量、通过数量、淘汰数量。
- `selected` 表示最终入选股票，统一包含股票代码、名称、板块信息和最新 K 线。
- `latest_kline` 字段保持 SQLite `daily_kline` 表字段口径，不为了输出改动原始数据口径。

## 技术方案

选股流程拆成两段：

1. 筛选阶段只准备过滤必须使用的数据。
2. 得到最终入选股票后，再补齐输出 JSON 所需的数据。

资金热度 `heat` 中，以下字段属于筛选必要字段，仍然需要在过滤前准备：

- `name`：用于 ST / 退市过滤。
- `turnover_rate`：用于换手率过滤。
- `price`：用于价格区间过滤。
- `amount`：用于成交额过滤，继续使用万元口径。
- `volume_ratio`：用于量比过滤。
- `change_pct`：用于涨幅过滤。

以下字段适合在选股结束后，只对最终入选股票补齐：

- `sectors`：调用 `data.get_relation(stock_code)` 获取所属板块，再复用 `business.map_relation_type()` 只保留行业和概念板块，避免对全部 A 股逐只查询导致流程过慢。
- `latest_kline`：从已经读取好的 `kline_map[stock_code][-1]` 取得，不需要再次查 SQLite。
- 其他展示字段：后续如果需要行业名称、概念列表、更多个股资料，都放在选后补齐阶段。

新增统一结果组装函数，所有策略最终都走同一个函数生成 JSON：

- `build_screen_result()`：组装统一的顶层结构。
- `build_screen_conditions()`：为每个策略生成条件列表。
- `build_selected_stock_items()`：为最终入选股票补齐名称、板块、最新 K 线。
- `build_stock_sectors()`：把通达信板块关系转换成统一的 `sectors` 数组，类型映射复用 `business.map_relation_type()`。

`conditions` 与 `layers` 一一对应，例如资金热度：

```json
[
  {
    "layer": "ST/退市过滤",
```

```python
def build_heat_conditions(

def normalize_latest_kline(kline_row):
    """按 SQLite daily_kline 字段口径输出最新 K 线。"""
    if not kline_row:
        return {}

    return {
        "code": kline_row.get("code", ""),
        "trade_date": kline_row.get("trade_date", ""),
        "open_price": safe_float(kline_row.get("open_price")),
        "high_price": safe_float(kline_row.get("high_price")),
        "low_price": safe_float(kline_row.get("low_price")),
        "close_price": safe_float(kline_row.get("close_price")),
        "volume": safe_float(kline_row.get("volume")),
        "amount": safe_float(kline_row.get("amount")),
    }


def build_stock_sectors(stock_code):
    """把通达信板块关系转换为统一 sectors 结构，只保留行业和概念。"""
    relation_rows = data.get_relation(stock_code)
    sectors = []
    seen = set()

    if not relation_rows:
        return sectors

    for row in relation_rows:
        if not isinstance(row, dict):
            continue

        sector_type = map_relation_type(row.get("BlockType"))
        if sector_type not in {"industry", "concept"}:
            continue

        sector_name = str(row.get("BlockName", "") or "").strip()
        sector_code = str(row.get("BlockCode", "") or "").strip()
        dedupe_key = (sector_type, sector_name, sector_code)
        if not sector_name or dedupe_key in seen:
            continue
        seen.add(dedupe_key)

        sectors.append(
            {
                "type": sector_type,
                "name": sector_name,
                "code": sector_code,
            }
        )

    return sectors


def build_selected_stock_items(stock_codes, kline_map, metric_map=None):
    """只为最终入选股票补齐统一 selected 输出。"""
    metric_map = metric_map or {}
    selected = []

    for stock_code in stock_codes:
        metrics = metric_map.get(stock_code, {})
        kline_bars = kline_map.get(stock_code, [])
        latest_kline = kline_bars[-1] if kline_bars else {}

        selected.append(
            {
                "code": stock_code,
                "name": metrics.get("name", ""),
                "sectors": build_stock_sectors(stock_code),
                "latest_kline": normalize_latest_kline(latest_kline),
            }
        )

    return selected


def build_screen_result(
    strategy,
    run_date,
    mode,
    conditions,
    selected,
    layers,
    initial_count=0,
    elapsed_seconds_value=0,
):
    """构造统一选股 JSON 输出。"""
    selected_codes = [item.get("code", "") for item in selected if item.get("code")]

    return {
        "strategy": strategy,
        "run_date": run_date or "",
        "mode": mode or "",
        "conditions": conditions,
        "selected": selected,
        "layers": layers,
        "initial_count": int(initial_count),
        "selected_count": len(selected),
        "selected_codes": selected_codes,
        "selected_items": selected,
        "elapsed_seconds": elapsed_seconds_value,
    }
```
